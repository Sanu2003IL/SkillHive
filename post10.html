1. String Indexing Recap

word = "PYTHON"
print(word[0])   # P
print(word[5])   # N
print(word[-1])  # N
print(word[-6])  # P

2. String Slicing Recap

s = "Programming"

print(s[0:6])    # Progra
print(s[:6])     # Progra
print(s[3:])     # gramming
print(s[:])      # Programming
print(s[::2])    # Pormig
print(s[::-1])   # gnimmargorP

3. Advanced String Methods

Searching

text = "Python is powerful and easy to learn."

print(text.find("powerful"))   # 10
print(text.find("Java"))       # -1
print(text.index("easy"))      # 22

Counting Occurrences

msg = "banana banana mango"

print(msg.count("banana"))   # 2
print(msg.count("a"))        # 6

Replacing

text = "I like Java"
print(text.replace("Java", "Python"))

Case Transformations

word = "python"
print(word.upper())
print(word.lower())
print(word.title())
print(word.capitalize())
print("HELLO".casefold())

Checking Properties

s = "Python3"
print(s.isalpha())
print("Python".isalpha())
print("123".isdigit())
print("hello".islower())
print("HELLO".isupper())
print("Hello World".istitle())

Whitespace Handling

txt = "   Python   "
print(txt.strip())
print(txt.lstrip())
print(txt.rstrip())

4. String Formatting

# Old Style
name = "Alice"
age = 25
print("My name is %s and I am %d years old." % (name, age))

# str.format()
name = "Bob"
lang = "Python"
print("Hello, my name is {} and I code in {}.".format(name, lang))

# f-Strings
name = "Charlie"
score = 95.5
print(f"My name is {name} and I scored {score}.")

5. Splitting & Joining

sentence = "Python is fun to learn"

words = sentence.split()  
print(words)

joined = "-".join(words)
print(joined)

6. Escape Characters

print("Hello\nWorld")
print("Hello\tWorld")
print("He said: \"Python is great!\"")
print("C:\\Users\\Name")

7. Raw Strings

path = r"C:\Users\Name\Documents"
print(path)

8. Multi-line Strings

msg = """This is a
multi-line string
in Python."""
print(msg)
