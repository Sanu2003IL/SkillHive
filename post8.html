1. Function Parameters & Arguments

1.1 Types of Arguments

1. Positional Arguments


def greet(name, age):
    print(f"Hello {name}, you are {age} years old.")

greet("Rahul", 21)   # Order matters
greet(21, "Rahul")   # Wrong order

2. Keyword Arguments


def greet(name, age):
    print(f"Hello {name}, you are {age} years old.")

greet(age=21, name="Rahul")  # Order doesn‚Äôt matter

3. Default Arguments


def greet(name, age=18):
    print(f"Hello {name}, you are {age} years old.")

greet("Rahul")        # Uses default age=18
greet("Ramesh", 25)   # Overrides default

4. Variable-length Arguments (*args)


def add_numbers(*args):
    total = sum(args)
    print("Sum:", total)

add_numbers(10, 20, 30, 40)   # Sum: 100

5. Variable-length Keyword Arguments (**kwargs)


def display_info(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

display_info(name="Rahul", age=21, city="Delhi")
# name: Rahul
# age: 21
# city: Delhi

2. Returning Multiple Values


def calculate(a, b):
    sum_val = a + b
    diff_val = a - b
    return sum_val, diff_val

s, d = calculate(10, 5)
print("Sum:", s)
print("Difference:", d)

3. Nested Functions


def outer():
    print("This is the outer function.")

    def inner():
        print("This is the inner function.")

    inner()

outer()

4. Lambda Functions

Lambda = small, one-line anonymous function.


square = lambda x: x * x
print(square(5))   # 25

add = lambda a, b: a + b
print(add(10, 20)) # 30

5. Map, Filter, Reduce with Lambda

map()


numbers = [1, 2, 3, 4, 5]
squares = list(map(lambda x: x**2, numbers))
print(squares)  # [1, 4, 9, 16, 25]

filter()


numbers = [10, 15, 20, 25, 30]
evens = list(filter(lambda x: x % 2 == 0, numbers))
print(evens)  # [10, 20, 30]

reduce()


from functools import reduce

numbers = [1, 2, 3, 4, 5]
product = reduce(lambda x, y: x * y, numbers)
print(product)  # 120

6. Recursion in Functions


def factorial(n):
    if n == 0 or n == 1:
        return 1
    else:
        return n * factorial(n - 1)

print(factorial(5))  # 120

‚ö†Ô∏è Note: Always include a base case to avoid infinite recursion.

7. Scope of Variables


x = 10  # Global variable

def func():
    x = 5  # Local variable
    print("Local x:", x)

func()
print("Global x:", x)

8. Global Keyword


count = 0

def increment():
    global count
    count += 1
    print("Count:", count)

increment()
increment()
